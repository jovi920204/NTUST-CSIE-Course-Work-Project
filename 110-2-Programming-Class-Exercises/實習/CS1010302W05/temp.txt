#include <iostream>
#include <vector>
using namespace std;

bool checkSudoku(vector<vector<int>> mySudoku);

int main() {
	char integer[9][18];
	while (cin.getline(integer[0], 18)) {
		bool answerFlag = true;
		vector<vector<int>> mySudoku(9);
		for (int i = 1; i < 9; i++) {
			cin.getline(integer[i], 18);
		}
		for (int i = 0; i < 9; i++) {
			for (int j = 0; j < 18; j++) {
				if (integer[i][j] == ',') {
					continue;
				}
				else if (isdigit(integer[i][j]) && !isdigit(integer[i][j++])) {
					mySudoku[i].push_back(integer[i][j]);
				}
				else {
					answerFlag = false;
					cout << "QQ" << endl;
					goto end;
				}
			}
		}
		answerFlag = checkSudoku(mySudoku);

	end:;
		if (answerFlag == true){
			cout << "True" << endl;
		}
		else {
			cout << "False" << endl;
		}
		cin.ignore();
	}
}

bool checkSudoku(vector<vector<int>> mySudoku) {
	for (int i = 0; i < 9; i++) {
		int check[10] = { 0 };
		for (int j = 0; j < 9; j++) {
			if (check[mySudoku[i][j]] == 1) {
				return false;
			}
			else {
				check[mySudoku[i][j]]++;
			}
		}
	}

	for (int i = 0; i < 9; i++) {
		int check[10] = { 0 };
		for (int j = 0; j < 9; j++) {
			if (check[mySudoku[j][i]] == 1) {
				return false;
			}
			else {
				check[mySudoku[j][i]]++;
			}
		}
	}

	for (int r = 0; r < 3; r++) {
		for (int c = 0; c < 3; c++) {
			int check[10] = { 0 };
			for (int i = 0; i < 3; i++) {
				for (int j = 0; j < 3; j++) {
					if (check[mySudoku[j + r * 3][i + c * 3]] == 1) {
						return false;
					}
					else {
						check[mySudoku[j + r * 3][i + c * 3]]++;
					}
				}
			}
		}
	}

	return true;
}
